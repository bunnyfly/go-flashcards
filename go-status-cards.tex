%%% \makeflashcardwithboards
%%%   1 - {pattern name}
%%%   2 - {question}
%%%   3 - {back text}
%%%   4 - {psgo board size}
%%%   5 - {psgo stone pattern}
%%%   6 - {answer 1 text}
%%%   7 - {answer 1 moves [defaults to black first]}
%%%   8 - {answer 2 text}
%%%   9 - {answer 2 moves [defaults to black first]}
\newcommand{\makeflashcardwithboards}[9]{%
  \makeflashcard{%
    %%% FRONT
    \setlength{\goxunit}{.4cm}\setlength{\goyunit}{.4cm}%
    \begin{psgopartialboard*}{#4}\normalsize#5\end{psgopartialboard*}%
    \\\LARGE#2%
  }{%
    %%% BACK
    \LARGE#1\vfill\normalsize#3\vfill%
    \ifx&#9&%
      \ifx&#7&%
        % 0 boards
      \else%
        % 1 board
        \begin{minipage}{\textwidth} \center%
          \begin{psgopartialboard*}{#4} \scriptsize \setcounter{gomove}{0} \setboolean{blackmove}{true}
            #5 #7
          \end{psgopartialboard*}\normalsize\\#6%
        \end{minipage}%
      \fi%
    \else%
      % 2 boards
      \begin{minipage}{0.5\textwidth} \center%
        \begin{psgopartialboard*}{#4} \scriptsize \setcounter{gomove}{0} \setboolean{blackmove}{true}
          #5 #7
        \end{psgopartialboard*}\normalsize\\#6%
      \end{minipage}%
      \begin{minipage}{0.5\textwidth} \center%
        \begin{psgopartialboard*}{#4} \scriptsize \setcounter{gomove}{0} \setboolean{blackmove}{true}
          #5 #9
        \end{psgopartialboard*}\normalsize\\#8%
      \end{minipage}%
    \fi%
  }
}
